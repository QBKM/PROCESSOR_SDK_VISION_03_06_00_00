883  }
884  else

887  pCPDMAPacketQueueDest->pTail->pNextPacket = pCPDMAPacketQueueSrc->pHead;
888  pCPDMAPacketQueueDest->count += pCPDMAPacketQueueSrc->count;
889  }
890  pCPDMAPacketQueueDest->pTail = pCPDMAPacketQueueSrc->pTail;
